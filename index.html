<script>

// ── STATE ──────────────────────────────────────

let businesses = {};  // { id: { name, icon, owner, ventas:[], gastos:[], servicios:[] } }

let currentBiz = null;

let quickSelected = null;


function save() { localStorage.setItem('multi_biz', JSON.stringify(businesses)); }

function load() {

  const d = localStorage.getItem('multi_biz');

  if (d) businesses = JSON.parse(d);

}

function biz() { return businesses[currentBiz]; }

function today() { return new Date().toISOString().split('T')[0]; }

function fmt(n) { return 'S/'+parseFloat(n||0).toFixed(2); }

function mkid() { return Date.now().toString(36)+Math.random().toString(36).slice(2,6); }

function monthKey(d) { return (d||'').slice(0,7); }

function fmtMes(m) {

  if (!m) return '';

  const [y,mo] = m.split('-');

  return ['Ene','Feb','Mar','Abr','May','Jun','Jul','Ago','Sep','Oct','Nov','Dic'][+mo-1]+' '+y;

}


// ── NEGOCIO SELECTOR ───────────────────────────

function renderBizScreen() {

  const grid = document.getElementById('biz-grid');

  const cards = Object.entries(businesses).map(([id,b]) => {

    const ing = b.ventas.reduce((a,v)=>a+parseFloat(v.monto||0),0);

    const gas = b.gastos.reduce((a,g)=>a+parseFloat(g.monto||0),0);

    return `<div class="biz-card" onclick="openBiz('${id}')">

      <div class="icon">${b.icon||'🏪'}</div>

      <div class="name">${b.name}</div>

      <div class="sub">${b.owner||''}</div>

      <div style="margin-top:8px;font-size:.75rem;color:#2d6a4f;font-weight:600">Gan: ${fmt(ing-gas)}</div>

    </div>`;

  }).join('');

  grid.innerHTML = cards + `<div class="biz-add" onclick="openModal('modal-new-biz')">➕<div style="font-size:.8rem;margin-top:4px;color:#2d6a4f">Nuevo negocio</div></div>`;

}


function openBiz(id) {

  currentBiz = id;

  const b = biz();

  document.getElementById('app-title').textContent = b.icon+' '+b.name;

  document.getElementById('app-owner').textContent = b.owner||'';

  document.getElementById('biz-screen').style.display = 'none';

  document.getElementById('app-screen').style.display = 'block';

  goTo('dashboard', document.querySelector('nav button'));

}


function goBack() {

  document.getElementById('app-screen').style.display = 'none';

  document.getElementById('biz-screen').style.display = 'flex';

  renderBizScreen();

}


function createBiz() {

  const name = document.getElementById('nb-name').value.trim();

  const icon = document.getElementById('nb-icon').value||'🏪';

  const owner = document.getElementById('nb-owner').value.trim();

  if (!name) { alert('Ingresa el nombre del negocio'); return; }

  const id = mkid();

  businesses[id] = { name, icon, owner, ventas:[], gastos:[], servicios:[] };

  save();

  closeModal('modal-new-biz');

  document.getElementById('nb-name').value='';

  document.getElementById('nb-owner').value='';

  renderBizScreen();

  openBiz(id);

}


// ── MODALS ─────────────────────────────────────

function openModal(id) { document.getElementById(id).classList.add('open'); }

function closeModal(id) { document.getElementById(id).classList.remove('open'); }


// ── NAV ────────────────────────────────────────

function goTo(page, btn) {

  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));

  document.querySelectorAll('nav button').forEach(b=>b.classList.remove('active'));

  document.getElementById(page).classList.add('active');

  if (btn) btn.classList.add('active');

  if (page==='dashboard') renderDashboard();

  if (page==='quick') renderQuick();

  if (page==='gastos') renderGastos();

  if (page==='servicios') renderServicios();

}


// ── DASHBOARD ──────────────────────────────────

function renderDashboard() {

  const b = biz();

  const sel = document.getElementById('mes-sel');

  const months = new Set([today().slice(0,7)]);

  b.ventas.forEach(v=>months.add(monthKey(v.fecha)));

  b.gastos.forEach(g=>months.add(monthKey(g.fecha)));

  const sorted = [...months].filter(Boolean).sort().reverse();

  const cur = sel.value || sorted[0];

  sel.innerHTML = sorted.map(m=>`<option value="${m}" ${m===cur?'selected':''}>${fmtMes(m)}</option>`).join('');

  const mes = sel.value || sorted[0];

  const ventas = b.ventas.filter(v=>monthKey(v.fecha)===mes);

  const gastos = b.gastos.filter(g=>monthKey(g.fecha)===mes);

  const ing = ventas.reduce((a,v)=>a+parseFloat(v.monto||0),0);

  const gas = gastos.reduce((a,g)=>a+parseFloat(g.monto||0),0);

  document.getElementById('s-ing').textContent=fmt(ing);

  document.getElementById('s-gas').textContent=fmt(gas);

  const ganEl=document.getElementById('s-gan');

  ganEl.textContent=fmt(ing-gas);

  ganEl.className='val '+(ing-gas>=0?'green':'red');

  const all=[...ventas.map(v=>({...v,tipo:'ingreso'})),...gastos.map(g=>({...g,tipo:'gasto'}))].sort((a,b)=>b.fecha.localeCompare(a.fecha));

  const tb=document.getElementById('mov-table');

  if (!all.length){tb.innerHTML='<tr><td colspan="4" class="empty">Sin movimientos</td></tr>';return;}

  tb.innerHTML=all.map(m=>`<tr>

    <td>${m.fecha}</td>

    <td>${m.tipo==='ingreso'?(m.servicio||m.descripcion||'-'):(m.descripcion||'-')}</td>

    <td><span class="badge ${m.tipo==='ingreso'?'bg':'br'}">${m.tipo==='ingreso'?'Venta':'Gasto'}</span></td>

    <td style="font-weight:600;color:${m.tipo==='ingreso'?'#2d6a4f':'#c0392b'}">${m.tipo==='ingreso'?'+':'-'}${fmt(m.monto)}</td>

  </tr>`).join('');

}


// ── VENTA RÁPIDA ───────────────────────────────

function renderQuick() {

  const b = biz();

  const grid = document.getElementById('quick-grid');

  const noSvc = document.getElementById('no-svc');

  if (!b.servicios.length) { grid.style.display='none'; noSvc.style.display='block'; }

  else {

    grid.style.display='grid'; noSvc.style.display='none';

    grid.innerHTML = b.servicios.map(s=>`

      <div class="quick-btn" onclick="openQuick('${s.id}')">

        <div class="qname">${s.nombre}</div>

        <div class="qprice">${s.precio?fmt(s.precio):'Precio libre'}</div>

      </div>`).join('');

  }

  renderVentas();

}


function openQuick(svcId) {

  const s = biz().servicios.find(s=>s.id===svcId);

  if (!s) return;

  quickSelected = s;

  document.getElementById('q-title').textContent = '⚡ '+s.nombre;

  document.getElementById('q-svc').textContent = s.nombre;

  document.getElementById('q-monto').value = s.precio||'';

  document.getElementById('q-cliente').value = '';

  document.getElementById('q-fecha').value = today();

  openModal('modal-quick');

}


function confirmQuick() {

  const monto = document.getElementById('q-monto').value;

  const cliente = document.getElementById('q-cliente').value;

  const fecha = document.getElementById('q-fecha').value || today();

  if (!monto||parseFloat(monto)<=0){alert('Ingresa un monto');return;}

  biz().ventas.push({id:mkid(),fecha,servicio:quickSelected.nombre,cliente,monto,nota:''});

  save(); closeModal('modal-quick'); renderQuick();

}


function addVentaManual() {

  const desc=document.getElementById('vm-desc').value.trim();

  const monto=document.getElementById('vm-monto').value;

  const fecha=document.getElementById('vm-fecha').value||today();

  const cliente=document.getElementById('vm-cliente').value;

  if (!monto||parseFloat(monto)<=0){alert('Ingresa un monto');return;}

  biz().ventas.push({id:mkid(),fecha,servicio:desc||'Servicio',cliente,monto,nota:''});

  save();

  document.getElementById('vm-desc').value='';

  document.getElementById('vm-monto').value='';

  document.getElementById('vm-cliente').value='';

  renderVentas();

}


function renderVentas() {

  const tb=document.getElementById('v-table');

  const sorted=[...biz().ventas].sort((a,b)=>b.fecha.localeCompare(a.fecha)).slice(0,20);

  if (!sorted.length){tb.innerHTML='<tr><td colspan="5" class="empty">Sin ventas</td></tr>';return;}

  tb.innerHTML=sorted.map(v=>`<tr>

    <td>${v.fecha}</td><td>${v.servicio||'-'}</td><td>${v.cliente||'-'}</td>

    <td style="color:#2d6a4f;font-weight:600">${fmt(v.monto)}</td>

    <td><button class="del-btn" onclick="delVenta('${v.id}')">🗑</button></td>

  </tr>`).join('');

}


function delVenta(id){

  if(!confirm('¿Eliminar?'))return;

  biz().ventas=biz().ventas.filter(v=>v.id!==id);

  save();renderVentas();

}


// ── GASTOS ─────────────────────────────────────

function addGasto() {

  const fecha=document.getElementById('g-fecha').value||today();

  const categoria=document.getElementById('g-cat').value;

  const monto=document.getElementById('g-monto').value;

  const descripcion=document.getElementById('g-desc').value;

  if (!monto||parseFloat(monto)<=0){alert('Ingresa un monto');return;}

  biz().gastos.push({id:mkid(),fecha,categoria,monto,descripcion});

  save();

  document.getElementById('g-fecha').value='';

  document.getElementById('g-monto').value='';

  document.getElementById('g-desc').value='';

  renderGastos();

}


function renderGastos() {

  const tb=document.getElementById('g-table');

  const sorted=[...biz().gastos].sort((a,b)=>b.fecha.localeCompare(a.fecha));

  if (!sorted.length){tb.innerHTML='<tr><td colspan="5" class="empty">Sin gastos</td></tr>';return;}

  tb.innerHTML=sorted.map(g=>`<tr>

    <td>${g.fecha}</td>

    <td><span class="badge br">${g.categoria}</span></td>

    <td>${g.descripcion||'-'}</td>

    <td style="color:#c0392b;font-weight:600">${fmt(g.monto)}</td>

    <td><button class="del-btn" onclick="delGasto('${g.id}')">🗑</button></td>

  </tr>`).join('');

}


function delGasto(id){

  if(!confirm('¿Eliminar?'))return;

  biz().gastos=biz().gastos.filter(g=>g.id!==id);

  save();renderGastos();

}


// ── SERVICIOS ──────────────────────────────────

function addServicio() {

  const nombre=document.getElementById('s-nombre').value.trim();

  const precio=document.getElementById('s-precio').value;

  const duracion=document.getElementById('s-dur').value;

  if (!nombre){alert('Ingresa el nombre');return;}

  biz().servicios.push({id:mkid(),nombre,precio,duracion});

  save();

  document.getElementById('s-nombre').value='';

  document.getElementById('s-precio').value='';

  document.getElementById('s-dur').value='';

  renderServicios();

}


function renderServicios() {

  const tb=document.getElementById('s-table');

  if (!biz().servicios.length){tb.innerHTML='<tr><td colspan="4" class="empty">Sin servicios</td></tr>';return;}

  tb.innerHTML=biz().servicios.map(s=>`<tr>

    <td style="font-weight:600">${s.nombre}</td>

    <td style="color:#2d6a4f">${s.precio?fmt(s.precio):'-'}</td>

    <td>${s.duracion||'-'}</td>

    <td><button class="del-btn" onclick="delServicio('${s.id}')">🗑</button></td>

  </tr>`).join('');

}


function delServicio(id){

  if(!confirm('¿Eliminar?'))return;

  biz().servicios=biz().servicios.filter(s=>s.id!==id);

  save();renderServicios();

}
